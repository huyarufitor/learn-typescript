# 02 Do you know Generic of ts?

# 在面向对象语言中，用来创建可复用的工具，我们称之为范型。利用范型，我们可以创建一个支持众多类型的组件，这让用户可以使用自己的类型消费（consume）这些组件

# - 1.identity function 恒等函数（传入什么内容，就返回什么内容）

function identity(argument:number):number{
  return arg;
}
// 这里是说，传入的是argument 数字类型，返回的可以很明确也是数字类型

或使用any 类型

function identity(argument:any): any {
  return argument;
}
//接受参数时可以接受任何类型，但是在返回的时候丢失了函数类型

所以我们需要一种可以捕获参数类型的方式，然后再用它表示返回值的类型，这时，这个恒等函数就是一个范型。

    1. 定义范型函数
function identity<Type>(argument:Type):Type{
    return argument;
}
//这个Type 是一个类型变量，允许我们捕获用户提供的类型。

  • 2.1 调用方法一

    let output = identity<string>("myString");
 // output: string

  • 2.2 调用方法二：类型参数推断

let output = identity("myString");
// 利用参数类型推断，可以不用<>明确传入类型，编译器看到myString，就会自动设置Type为mystring的类型--string

# 2. 使用范型类型变量

# 下方函数，如果想打印参数的lenth，是会报错，因为不知道传入的Type类型是否有length属性

function loggingIdentity<Type>(arg:Type):Type{
   console.log(arg.length);
   // // Property 'length' does not exist on type 'Type'.
   return arg;
}

---- 解决办法： 范型函数loggingIdentity接受一个Type的类型参数和一个实参arg,实参arg是一个Type类型的数组。函数返回的是Type类型数组，这样，使用的是类型的一部分，灵活度++；

# here are new code

function loggingIdentity<Type>(arg:Typep[]):Type[]{
    console.log(arg.length);
    return arg;
}

# 3.范型接口（范型类型 Generic Types）

  • 范型函数1
function identity<Type>(arg:Type):Type{
    retuan arg;
}
let myIdentity:<Type>(arg:Type) => Type = identity;
  • 范型函数2
function identity<Type>(arg:Type): Type{
    return arg;
}
let myIdentity:<Input>(arg:Input) => Input = identity;
  • 范型函数3: 以对象类型的调用签名的形式，书写范型类型；
function identity<Type>(arg:Type): Type{
    return arg;
}
let myIdentity: { <Type>(arg:Type):Type } = identity;

# 所以有了范型接口

// 范型接口GenericIdentityFn
interface GenericIdentityFn{
    <Type>(arg:Type):Type;
}
function identity<Type>(arg:Type):Type{
    return arg;
}
let myIdentity: GenericIdentityFn = identity;

有时候希望在使用范型的时候就可以看到对应的参数，即将范型参数作为整个接口的参数，可以这样写：

interface GenericIdentityFn<Type>{
    (arg:Type): Type;
}
function identity<Type>(arg:Type): Type{
    return arg;
}
let myIdentity:GenericIdentityFn<number> = identity;
# 除了范型接口之外，还可以创建范型类。但是不能创建范型枚举类型和范型命名空间

# 范型类

# （1）写法：类似范型接口，在类名后面，使用<>包裹住类型参数列表

// 定义接口
class GenericNumber<NumType>{
    zegoValue: NumType;
    add:(x:NumType,y:NumType) => NumType;
}
let myGenericNumber = new GenericNumber<number>();
myGenericNumber.zegoValue = 0;
myGenericNumber.add  = function(x,y){
    return x+y;
}

# 上面函数，没有规定只能用number作为NumType，可以用其他的基础类型或者复杂类型

# 总结： class 类有两部分：静态部分和实例部分。范型类仅仅对实例部分生效，所有当我们使用类的时候，注意 静态成员不能使用类型参数
